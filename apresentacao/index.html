Javascript?

  • Não é Java
    • Sintaxe parecida
    • Tipagem fraca
    • Sem tipos estáticos
  • Também conhecido como ECMAScript
  • Multiplataforma
  • Multiparadigma
    • Funções de primeira classe
      • Função é um tipo linguagem
      • Orientado a objetos
        • "Prototype-based", ao invés de "class based"1
      • Funcional
      • Imperativo
1 - A versão 6 do ECMAScript adiciona suporte a classes

new e this


/**
 * new: Constrói um objeto a partir de uma função
 * this: Referência ao objeto atual

typeof


/**
 * typeof: Retorna uma string do tipo não avaliado do operando.
 *         Os valores possíveis são: "function", "string", "number",
 *         "object" ou "undefined"
 */
console.log(typeof (function(){})); //escreve "function"
console.log(typeof "Hello world"); //escreve "string"
console.log(typeof 1); //escreve "number"
console.log(typeof {}); //escreve "object"
console.log(typeof undefined); //escreve "undefined"

instanceof


/**
 * instanceof: Retorna true se o objeto é uma instância da função
 *             requisitada
 */
var agora = new Date();
console.log(agora instanceof Date); //Escreve true
console.log(agora instanceof Object); //Escreve true
console.log(agora instanceof Array); //Escreve false
var vetor = [];
console.log(vetor instanceof Date); //Escreve false
console.log(vetor instanceof Object); //Escreve true
console.log(vetor instanceof Array); //Escreve true

console.log(typeof Date);//Escreve "function"
console.log(typeof Object);//Escreve "function"
console.log(typeof Array);//Escreve "function"

void


/**
* void: Executa a expressão, mas não retorna valor
*/
function voidtest() {
  console.log("Teste void!");
  return "teste";
}
var teste = void(voidtest()); //Escreve "Teste void!"
console.log(teste); //Escreve undefined

while


//while(condição)
var vetor = [2,3,5,7,11];
var i = 0, len = vetor.length;
while(i < len) {
  console.log("vetor[", i, "] = ", vetor[i]);
  i++;
}

do while


/**
 * do {
 *   instruções
 * } while( condição )
 */
var vetor = [2,3,5,7,11];
var i = 0, len = vetor.length;
do {
  console.log("vetor[", i, "] = ", vetor[i]);
  i++;
} while(i < len);

for


//for (inicialização; condição; expressão)
var vetor = [2,3,5,7,11];
for(var i = 0, len = vetor.length; i < len; i++) {
  console.log("vetor[", i, "] = ", vetor[i]);
}

for in


//for (atributo in objeto)
var obj = {
  "nome": "Gabriel",
  "sobrenome": "Garcia Marquez"
};
for(var atributo in obj) {
  console.log("obj.", atributo, " = ", obj[atributo]);
}

Rótulos, continue e break


var vetor = [2,3,5,7,11];
var i = 0, len = vetor.length;
repetir: //Isto é um rótulo (label)
for(;;){
  console.log(vetor[i]);
  i++;
  if (i < len) {
    continue repetir;//continue: usado dentro de loops
  }
  //break: Sai de um switch, rótulo ou loop
  break;
}
console.log("Fim, mas não use este código!");
